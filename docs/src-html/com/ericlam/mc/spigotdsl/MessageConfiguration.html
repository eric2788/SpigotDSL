001package com.ericlam.mc.spigotdsl;
002
003
004import org.bukkit.ChatColor;
005import org.bukkit.configuration.file.YamlConfiguration;
006import org.jetbrains.annotations.NotNull;
007
008import java.util.List;
009
010/**
011 * a FileConfiguration for message that auto translate the color and add the prefix
012 */
013public abstract class MessageConfiguration extends YamlConfiguration {
014
015    /**
016     * translate color
017     * @param msg text to translate
018     * @return if msg is null then it will return empty string
019     */
020    @NotNull
021    public static String translate(String msg) {
022        return ChatColor.translateAlternateColorCodes('&', msg != null ? msg : "");
023    }
024
025    /**
026     * get the prefix text with translated color
027     * @return prefix text
028     */
029    @NotNull
030    public abstract String getPrefix();
031
032    /**
033     * get string without color and prefix.
034     * @param path path
035     * @return string without color and prefix.
036     */
037    @NotNull
038    public abstract String getPureString(@NotNull String path);
039
040    /**
041     * get string list without color and prefix
042     * @param path path
043     * @return string list without color and prefix.
044     */
045    @NotNull
046    public abstract List<String> getPureList(@NotNull String path);
047
048}
