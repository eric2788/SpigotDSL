001package com.ericlam.mc.spigotdsl.api;
002
003import org.bukkit.event.Event;
004import org.bukkit.event.EventPriority;
005
006import java.util.function.Consumer;
007
008/**
009 * here to register all events
010 */
011public interface EventListeners {
012
013    /**
014     * fast register an event
015     * @param event event class name
016     * @param handler event handler
017     * @param <E> bukkit event
018     */
019    <E extends Event> void listen(Class<E> event, Consumer<E> handler);
020
021    /**
022     * register an event with builder
023     * @param event event class name
024     * @param <E> bukkit event
025     * @return event listen builder
026     */
027    <E extends Event> ListenBuilder<E> event(Class<E> event);
028
029    /**
030     * a builder to create event handler
031     * @param <E> bukkit event
032     */
033    interface ListenBuilder<E extends Event> {
034
035        /**
036         * set the priority for that event
037         * @param priority event priority
038         * @return builder
039         */
040        ListenBuilder<E> priority(EventPriority priority);
041
042        /**
043         * set whether to not execute when setCancelled is true
044         * @param ignoreCancelled set the ignoreCancelled
045         * @return builder
046         */
047        ListenBuilder<E> ignoreCancelled(boolean ignoreCancelled);
048
049        /**
050         * set the event handler
051         * @param handler event handler
052         */
053        void listen(Consumer<E> handler);
054
055    }
056
057}
